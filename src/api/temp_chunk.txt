export const aiGenerateImageWithProgress = async (
  prompt: string,
  onProgress?: (pct: number) => void,
  options?: { width?: number; height?: number; seed?: number; model?: ImageModelType }
): Promise<string> => {
  // Simulate progress
  if (onProgress) {
    let p = 0;
    const interval = setInterval(() => {
      p += 15;
      if (p >= 90) clearInterval(interval);
      onProgress(Math.min(p, 90));
    }, 500);
  }

  const { width = 1024, height = 1024, seed = Math.floor(Math.random() * 99999), model = 'flux' } = options || {};

  const headers = await createSignedHeaders({ prompt, model, width, height, seed });
  headers['Content-Type'] = 'application/json';

  const response = await fetch(`/api/generate-image`, {
    method: 'POST',
    headers,
    body: JSON.stringify({ prompt, model, width, height, seed })
  });

  if (!response.ok) {
    let errorMsg = 'Image generation failed';
    try {
      const errData = await response.json();
      if (errData.error) errorMsg = errData.error;
    } catch (e) {
      // Ignore parse error
    }
    throw new Error(errorMsg);
  }

  const blob = await response.blob();
  if (onProgress) onProgress(100);

  return new Promise<string>((resolve, reject) => {
    const reader = new FileReader();
    reader.onloadend = () => resolve(reader.result as string);
    reader.onerror = reject;
    reader.readAsDataURL(blob);
  });
};

// ─── Helper: Handle SSE Streaming ─────────────────────────────────
async function handleSSEStream(res: Response, onChunk: (c: string, citations?: string[]) => void): Promise<string> {
  const reader = res.body!.getReader();
  const decoder = new TextDecoder('utf-8');
  let fullReply = '';
  let buffer = '';
  let citations: string[] | undefined = undefined;

  let hasStartedThinking = false;
  let hasFinishedThinking = false;

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });

    let changed = false;
    let boundary = buffer.indexOf('\n');
    while (boundary !== -1) {
      const line = buffer.slice(0, boundary).trim();
      buffer = buffer.slice(boundary + 1);

      if (line.startsWith('data: ')) {
        const dataStr = line.slice(6).trim();
        if (dataStr === '[DONE]') continue;
        try {
          const parsed = JSON.parse(dataStr);
          if (parsed.citations && Array.isArray(parsed.citations)) citations = parsed.citations;

          const delta = parsed.choices?.[0]?.delta || {};
          const msgContent = delta.content || parsed.message || '';
          const reasoningContent = delta.reasoning_content || delta.reasoning || '';

          if (reasoningContent) {
            if (!hasStartedThinking) {
              fullReply += '<think>\n';
              hasStartedThinking = true;
            }
            fullReply += reasoningContent;
            changed = true;
          }

          if (msgContent) {
            if (hasStartedThinking && !hasFinishedThinking) {
              fullReply += '\n</think>\n\n';
              hasFinishedThinking = true;
            }
            fullReply += msgContent;
            changed = true;
          }
        } catch (e) {
          // Ignore partial JSON parsing errors if any
        }
      }
      boundary = buffer.indexOf('\n');
    }

    if (changed) {
      onChunk(fullReply, citations);
    }
  }

  if (hasStartedThinking && !hasFinishedThinking) {
    fullReply += '\n</think>\n';
  }

  return fullReply || 'No response received from the AI model.';
}

// ─── Provider Rate Limit Cache Helpers ────────────────────────────
function setProviderLimit(provider: string) {
  try {
    localStorage.setItem(`inixa_rate_limit_${provider}`, Date.now().toString());
  } catch (e) {
    // Ignore localStorage errors
  }
}

function checkProviderLimit(provider: string): boolean {
  try {
    const stored = localStorage.getItem(`inixa_rate_limit_${provider}`);
    if (!stored) return false;

    const timestamp = parseInt(stored, 10);
    const ONE_HOUR = 60 * 60 * 1000;

    if (Date.now() - timestamp < ONE_HOUR) {
      return true; // Limit is active
    } else {
      localStorage.removeItem(`inixa_rate_limit_${provider}`); // Expired
      return false;
    }
  } catch (e) {
    return false;
  }
}

// ─── Main Chat Engine ─────────────────────────────────────────────
// All requests go through our Next.js API route (/api/chat)
export const aiChat = async (
  messages: any[],
  onChunk?: (c: string, citations?: string[]) => void,
  modelOverride?: any
): Promise<string> => {
  try {
    const model = modelOverride || getSelectedModel();

    // Extract the text content from the last message
    const lastMessage = messages[messages.length - 1];

    // Sanitize user inputs
    const safeMessages = messages.map(m => {
      let safeContent = m.content;
      if (m.role === 'user' && typeof m.content === 'string') {
        safeContent = sanitizeInput(m.content);
      }
      return { ...m, content: safeContent };
    });

    // Build full conversation history for context
    const conversationHistory = safeMessages.map(m => ({
      role: m.role as string,
      content: typeof m.content === 'string'
        ? m.content
        : Array.isArray(m.content)
          ? m.content.filter((c: any) => c.type === 'text').map((c: any) => c.text).join('\n')
          : String(m.content)
    }));

    const modelStr = model.modelStr;
    const API_BASE = process.env.NEXT_PUBLIC_API_URL || '';

    // Route everything internally to Next.js API routes
    let endpointPath = '/api/chat/completions';
    if (model.engine === 'g4f' || modelStr.startsWith('g4f/') || modelStr.startsWith('qwen_worker/')) {
      endpointPath = '/api/chat/g4f';
    }
    const fetchUrl = `${API_BASE}${endpointPath}`;

    // --- Try direct client-side fetch first (User IP) ---
    try {
      console.log(`[Client Fetch] Attempting direct fetch for ${model.label}...`);
      const directResponse = await attemptDirectClientCall(modelStr, conversationHistory, !!onChunk);

      if (directResponse.ok) {
        console.log(`[Client Fetch] Direct fetch successful for ${model.label}!`);

        if (onChunk && directResponse.body) {
          return await handleSSEStream(directResponse, onChunk);
        }

        const data = await directResponse.json();
        const reply = data.choices?.[0]?.message?.content || data.reply || '';
        const reasoning = data.choices?.[0]?.message?.reasoning_content || data.choices?.[0]?.message?.reasoning || '';
        if (reasoning) {
          return `<think>\n${reasoning}\n</think>\n\n${reply}`;
        }
        if (reply) return reply;
      } else {
        console.warn(`[Client Fetch] Direct fetch returned ${directResponse.status}, falling back...`);
      }
    } catch (e) {
      console.warn(`[Client Fetch Failed] Falling back to backend proxy. Reason:`, e);
    }
    // ----------------------------------------------------

    const bodyData = {
      messages: conversationHistory,
      model: modelStr,
      provider: model.provider,
      stream: !!onChunk
    };

    const secureHeaders = await createSignedHeaders(bodyData);
    secureHeaders['Content-Type'] = 'application/json';

    const res = await fetch(fetchUrl, {
      method: 'POST',
      headers: secureHeaders,
      body: JSON.stringify(bodyData)
    });

    if (!res.ok) {
      if (res.status === 429) {
        return '⚠️ Rate limit exceeded. Please wait a moment and try again.';
      }
      try {
        const errorData = await res.json();
        if (errorData.reply) return errorData.reply;
        if (errorData.error) return `⚠️ API Error: ${errorData.error}`;
      } catch { }
      return `❌ Error: Server returned ${res.status}.`;
    }

    if (onChunk && res.body) {
      return await handleSSEStream(res, onChunk);
    }

    const data = await res.json();
    const reply = data.choices?.[0]?.message?.content || data.reply || '';
    const reasoning = data.choices?.[0]?.message?.reasoning_content || data.choices?.[0]?.message?.reasoning || '';
    if (reasoning) {
      return `<think>\n${reasoning}\n</think>\n\n${reply}`;
    }
    return reply || 'No response received from the AI model.';
  } catch (e) {
    console.error('Chat API Error', e);
    return '❌ Connection failed. Please try a different model or check your connection.';
  }
};

// ─── Web Search & Scrape ─────────────────────────────────────────────
export const aiWebSearch = async (query: string): Promise<any[]> => {
  try {
    const safeQuery = sanitizeInput(query);
    const bodyData = { query: safeQuery };
    const secureHeaders = await createSignedHeaders(bodyData);
    secureHeaders['Content-Type'] = 'application/json';

    const res = await fetch(`/api/web-search`, {
      method: 'POST',
      headers: secureHeaders,
      body: JSON.stringify(bodyData)
    });
    const data = await res.json();
    if (data.ok && data.results) return data.results;
  } catch (e) {
    console.error('Web Search Error:', e);
  }
  return [
